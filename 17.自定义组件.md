### 创建

在项目根目录创建`components`文件夹

在`components`文件夹下创建`test`文件夹

在`test`文件夹右键,新建Components

### 引用

- 全局引用

  ```
  app.json文件中引入
  与pages平级
  "usingComponents": {
      "ikun":"/components/ikun/ikun"
    }
  ```

  

- 局部引用

  ```
  在页面.json文件中引入
  {
  	"usingComponents":{
  		"my-test":"/components/test/test"
  	}
  }
  
  在wxml中使用组件
  <my-test></my-test>
  
  ```

  

### 样式

默认情况下自定义组件只会对当前组件生效,不会影响到组件之外的样式

`styleIsoLation`可修改组件效果

```
在组件 .js文件中新增配置
与components平级
options:{
    styleIsolation:'isolated'		//启用样式隔离
    //可选值
    //默认	isolated
    //apply-shared	组件可以影响外界
    //shared		外界可以与组件相互影响
}
```



### data和methods

```
data中可定义私有数据
data: {
    count:0		//count变量,初始值为0
}

methods

methods: {
    //点击事件函数
    addCount(){
      this.setData({
        count:this.data.count+1		//count值+1
      })
      this._showCount()	//调用显示事件
    },
    _showCount(){
      wx.showToast({		//弹出信息
        title:"count:"+this.data.count,	//返回页面count值
        icon:'none'			//无图标
      })
    }
  }
```



### properties

接收外界传输的数据

```
properties:{
    max:{			//完整定义
      type:Number,	//数据类型
      value:0		//默认值
    },
    max:Number		//简化定义
  },
  
<ikun max="10"></ikun>
```

通过properties限制数量

```
<ikun max="10"></ikun>

// components/ikun/ikun.js
Component({
  options:{
    styleIsolation:'isolated'
  },
  /**
   * 组件的属性列表
   */
  properties: {
    max:{
      type:Number,
      value:10
    }
  },

  /**
   * 组件的初始数据
   */
  data: {
    count:0
  },

  /**
   * 组件的方法列表
   */
  methods: {
    //点击事件函数
    addCount(){
      if(this.data.count >= this.properties.max) return
      this.setData({
        count:this.data.count+1
      })
      this._showCount()
    },
    _showCount(){
      wx.showToast({
        title:"count:"+this.data.count,
        icon:'none'
      })
    }
  }
})

```



### 数据侦听器

```
data: {
    num1:0,
    num2:0,
    sum:0
  },

  /**
   * 组件的方法列表
   */
  methods: {
    //点击事件函数
    addNum1(){
      this.setData({
        num1:this.data.num1+1
      })
    },
    addNum2(){
      this.setData({
        num2:this.data.num2+1,
      })
    },
    reNum1(){
      this.setData({
        num1:this.data.num1-1
      })
    },
    reNum2(){
      this.setData({
        num2:this.data.num2-1,
      })
    },
    observers:{
    'num1,num2':function (num1,num2) {
      this.setData({
        sum:num1+num2
      })
    }
  }
    
   
   
<view>{{num1}}+{{num2}}={{sum}}</view> 
<button bindtap="addNum1">+1</button> 
<button bindtap="addNum2">+1</button>
<button bindtap="reNum1">-1</button>
<button bindtap="reNum2">-1</button>
<button bindtap="sum">求和</button>
    
```

### 插槽

- 单个插槽

```
自定义插件.wxml
<slot></slot>	//对于不确定的内容，可以使用slot占位

<插件>
	slot内容
</插件>
```

- 多个插槽

```
在js文件中启用slot支持
options:{
	multipleSlots:'true'
}

//定义
<view>
  <slot name="one"></slot>
  <text>内部结构</text>
  <slot name="two" ></slot>
</view>

//调用
<组件>
<view slot="one">1</view>
<view slot="two">2</view>
<组件>
```
